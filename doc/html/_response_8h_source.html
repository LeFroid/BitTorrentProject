52  Response(const std::string &data)
53  {
54  std::size_t versionPos = data.find_first_of('/');
55  std::size_t statusPos = data.find_first_of(' ', versionPos);
56  // Extract version of protocol
57  if (versionPos != std::string::npos && statusPos != std::string::npos)
58  {
59  Version = data.substr(versionPos + 1, statusPos - versionPos - 1);
60 
61  // Extract status code, followed by reason
62  std::size_t reasonPos = data.find(' ', versionPos + 1);
63  if (reasonPos != std::string::npos)
64  {
65  try {
66  StatusCode = std::stoi(data.substr(statusPos + 1, reasonPos - statusPos - 1));
67  } catch (const std::invalid_argument &err) {
68  LOG_ERROR("torrent_protocol.http", "Error parsing status code from HTTP resonse, message: ", err.what());
69  }
70 
71  reasonPos = data.find(' ', statusPos + 1);
72  Reason = data.substr(reasonPos + 1, data.find("\r\n", reasonPos + 1) - reasonPos - 1);
73  }
74  }
75 
76  // Get position of end of header, extract payload
77  std::size_t headerEndPos = data.find("\r\n\r\n");
78  if (headerEndPos != std::string::npos && headerEndPos + 4 < data.size())
79  {
80  Payload = data.substr(headerEndPos + 4);
81  }
82  }
